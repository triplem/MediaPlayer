432 			if (resource_name.toUpperCase().endsWith("JPG")) {

435 			}

440 			writer.writeResourceEnd();

442 			log.error("Error Writing Resource: " + resource_name, e);

466 	private String getMacAddress() {
467 		String mac = "";
468 		String command = "/sbin/ifconfig";
469 
470 		String sOsName = System.getProperty("os.name");
471 		if (sOsName.startsWith("Windows")) {
472 			command = "ipconfig /all";
473 		} else {
474 
475 			if ((sOsName.startsWith("Linux")) || (sOsName.startsWith("Mac")) || (sOsName.startsWith("HP-UX"))) {
476 				command = "/sbin/ifconfig";
477 			} else {
478 				log.info("The current operating system '" + sOsName + "' is not supported.");
479 			}
480 		}
481 
482 		Pattern p = Pattern.compile("([a-fA-F0-9]{1,2}(-|:)){5}[a-fA-F0-9]{1,2}");
483 		try {
484 			Process pa = Runtime.getRuntime().exec(command);
485 			pa.waitFor();
486 			BufferedReader reader = new BufferedReader(new InputStreamReader(pa.getInputStream()));
487 
488 			String line;
489 			Matcher m;
490 			while ((line = reader.readLine()) != null) {
491 				log.debug(line);
492 				m = p.matcher(line);
493 
494 				if (!m.find())
495 					continue;
496 				line = m.group();
497 				mac = line.replace(":", "");
498 				mac = mac.replace("-", "");
499 				break;
500 
501 			}
502 			// System.out.println(line);
503 		} catch (Exception e) {
504 			log.debug("Error MAC Address: ", e);
505 		}
506 		return mac;
507 	}
