42  	private void createPins() {
43  		if (gpio == null)
44  			return;
45  		if(!default_pin.equalsIgnoreCase(""))
46  		{
47  			log.debug("Creating Default_Pin: " + default_pin);
48  			Pin pin_number = createPin(default_pin);
49  			if (pin_number != null) {
50  				GpioPinDigitalOutput pin = gpio.provisionDigitalOutputPin(pin_number, // PIN
51  						// NUMBER
52  						"default_pin", // PIN FRIENDLY NAME (optional)
53  						PinState.LOW); // PIN STARTUP STATE (optional)
54  				pins.put(default_pin, pin);
55  			}
56  		}
57  		log.debug("Creating Pins from Sources");
58  		for (String key : sources.keySet()) {
59  			Source s = sources.get(key);
60  			if (!pins.containsKey(s.getGPIO_PIN())) {
61  				Pin pin_number = createPin(s.getGPIO_PIN());
62  				if (pin_number != null) {
63  					GpioPinDigitalOutput pin = gpio.provisionDigitalOutputPin(pin_number, // PIN
64  							// NUMBER
65  							s.getName(), // PIN FRIENDLY NAME (optional)
66  							PinState.LOW); // PIN STARTUP STATE (optional)
67  					pins.put(s.getGPIO_PIN(), pin);
68  				}
69  			}
70  		}
71  	}

95  	private void updateSource(String name)
96  	{
97  		if (gpio == null)
98  			return;
99  		if (sources.containsKey(name)) {
100 			try {
101 				log.debug("Changed Source ClearPins");
102 				clearLEDS();
103 				Source source = sources.get(name);
104 				log.debug("Source PIN: " + source.getGPIO_PIN());
105 				if (pins.containsKey(source.getGPIO_PIN())) {
106 					log.debug("Changed Source Take Pin: " + source.getGPIO_PIN() + " High");
107 					GpioPinDigitalOutput pin = pins.get(source.getGPIO_PIN());
108 					pin.high();
109 				}
110 
111 			} catch (Exception e) {
112 				log.error("Error Setting Pins: " + e);
113 			}
114 		} else {
115 			log.debug("Not A CustomSource Change Input to PIN: " + default_pin);
116 			try
117 			{
118 			clearLEDS();
119 			GpioPinDigitalOutput pin = pins.get(default_pin);
120 			pin.high();		
121 			}
122 			catch(Exception e )
123 			{
124 				log.error("Error Making Default Pin High",e);
125 			}
126 		}
